    <script>
        // Bottom navigation functionality
        document.querySelectorAll('.nav-item').forEach(item => {
            item.addEventListener('click', () => {
                const tabId = item.dataset.tab;
                
                // Remove active class from all nav items and panels
                document.querySelectorAll('.nav-item').forEach(nav => nav.classList.remove('active'));
                document.querySelectorAll('.tab-panel').forEach(panel => panel.classList.remove('active'));
                
                // Add active class to selected nav item and panel
                item.classList.add('active');
                document.getElementById(tabId).classList.add('active');
            });
        });

        // Action buttons functionality
        document.querySelectorAll('.action-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const action = e.target.textContent;
                const card = e.target.closest('.table-card');
                const id = card.querySelector('.card-id').textContent;
                
                switch(action) {
                    case 'View':
                        alert(`Viewing record ${id}`);
                        break;
                    case 'Edit':
                        alert(`Editing record ${id}`);
                        break;
                    case 'Delete':
                    case 'Block':
                        if(confirm(`Are you sure you want to ${action.toLowerCase()} record ${id}?`)) {
                            alert(`Record ${id} has been ${action.toLowerCase()}ed`);
                        }
                        break;
                }
            });
        });

        // Logout functionality
        document.querySelector('.logout-btn').addEventListener('click', () => {
            if(confirm('Are you sure you want to logout?')) {
                alert('Logging out...');
            }
        });

        // Save profile functionality
        document.querySelector('.save-btn').addEventListener('click', () => {
            alert('Profile settings saved successfully!');
        });

        // Loading animation
        window.addEventListener('load', () => {
            document.querySelector('.mobile-container').style.opacity = '0';
            document.querySelector('.mobile-container').style.transform = 'translateY(20px)';
            
            setTimeout(() => {
                document.querySelector('.mobile-container').style.transition = 'all 0.5s ease';
                document.querySelector('.mobile-container').style.opacity = '1';
                document.querySelector('.mobile-container').style.transform = 'translateY(0)';
            }, 100);
        });
    </script>
