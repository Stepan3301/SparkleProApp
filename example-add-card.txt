    <script>
        function showCardForm() {
            document.getElementById('cardForm').classList.add('active');
            document.querySelector('.add-card-btn').style.display = 'none';
        }

        function hideCardForm() {
            document.getElementById('cardForm').classList.remove('active');
            document.querySelector('.add-card-btn').style.display = 'block';
            document.getElementById('successMessage').classList.remove('active');
            resetForm();
        }

        function resetForm() {
            document.getElementById('cardNumber').value = '';
            document.getElementById('cardName').value = '';
            document.getElementById('expiry').value = '';
            document.getElementById('cvvCard').value = '';
        }

        // Format card number input
        document.getElementById('cardNumber').addEventListener('input', function(e) {
            let value = e.target.value.replace(/\s/g, '').replace(/[^0-9]/gi, '');
            let formattedValue = value.match(/.{1,4}/g)?.join(' ') || value;
            e.target.value = formattedValue;
        });

        // Format expiry date
        document.getElementById('expiry').addEventListener('input', function(e) {
            let value = e.target.value.replace(/\D/g, '');
            if (value.length >= 2) {
                value = value.substring(0, 2) + '/' + value.substring(2, 4);
            }
            e.target.value = value;
        });

        // CVV input validation
        document.getElementById('cvvCard').addEventListener('input', function(e) {
            e.target.value = e.target.value.replace(/[^0-9]/g, '');
        });

        // Card name input
        document.getElementById('cardName').addEventListener('input', function(e) {
            if (e.target.value.length > 20) {
                e.target.value = e.target.value.substring(0, 20);
            }
        });

        function submitCard() {
            const cardNumber = document.getElementById('cardNumber').value;
            const cardName = document.getElementById('cardName').value;
            const expiry = document.getElementById('expiry').value;
            const cvv = document.getElementById('cvvCard').value;

            if (!cardNumber || !cardName || !expiry || !cvv) {
                alert('Please fill in all fields');
                return;
            }

            // Hide form and show success message
            document.getElementById('cardForm').classList.remove('active');
            document.getElementById('successMessage').classList.add('active');
            
            // Reset form after 3 seconds
            setTimeout(() => {
                document.getElementById('successMessage').classList.remove('active');
                document.querySelector('.add-card-btn').style.display = 'block';
                resetForm();
            }, 3000);
        }

        // Add focus effects
        const cardInputs = document.querySelectorAll('.card-number-input, .card-name-input, .card-expiry-input, .card-cvv-input');
        cardInputs.forEach(input => {
            input.addEventListener('focus', function() {
                this.style.backgroundColor = 'rgba(255, 255, 255, 0.1)';
                this.style.borderRadius = '8px';
                this.style.padding = '5px';
            });
            
            input.addEventListener('blur', function() {
                this.style.backgroundColor = 'transparent';
                this.style.padding = '0';
            });
        });
    </script>
